#include <mega32.h>
#include <delay.h>
unsigned char i=0;
void main(void)
{
 DDRA=0xff;
 DDRD.2=0;
 DDRD.3=0;
 DDRB.0=0;
 DDRD.7=1;
while (1)
    {
     if(PIND.2==0&&i<15)
      {
       i++;
       PORTA=i;
       PORTD.7=1;
       delay_ms(5);
       PORTD.7=0;
       delay_ms(200);
      }
     if(PIND.3==0&&i>0)
      {
       i--;
       PORTA=i;
       PORTD.7=1;
       delay_ms(5);
       PORTD.7=0;
       delay_ms(200);
      } 
      if(PINB.0==0)
      {
       i=0;
       PORTA=i;
       PORTD.7=1;
       delay_ms(100);
       PORTD.7=0;
      }

    }
}
