#include <mega32.h>
#include <delay.h>
unsigned char d=0,y=0;
void main(void)
{
 DDRA=0xff;
 DDRB=0xff;
while (1)
    {
     y++;
     if(y==10)
     {
      y=0;
      d++;
      if(d==10) d=0;
     }
     PORTA=d;
     PORTB=y;
     delay_ms(100);

    }
}

#include <mega32.h>
#include <delay.h>
unsigned char d=0,y=0;
unsigned char i;
void main(void)
{
 DDRA=0xff;
 DDRB=0xff;
while (1)
    {
     for(i=0;i<100;i++)
     {
      d=i/10;
      y=i%10;
      PORTA=d;
      PORTB=y;
      delay_ms(100);
     }

    }
}
